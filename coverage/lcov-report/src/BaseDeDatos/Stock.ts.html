/**
 * Representa una transaccin de compra, venta o devolucin de muebles.
 */
interface Transaction {
    id: number;
    type: 'SALE' | 'PURCHASE' | 'RETURNED_BY_CUSTOMER' | 'RETURN_TO_SUPPLIER';
    furnitureID: number;
    quantity: number;
    interactorID: number; // Cliente o Proveedor ID
    date: string;
    amount: number;
}
 
/**
 * Representa el esquema de la base de datos.
 */
interface DbSchema {
    sillas:Silla[];
    armarios:Armario[];
    mesas:Mesa[]
    proveedores: Proveedor[];
    clientes: Cliente[];
    stock: {
        [key: string]: { [id: number]: number }; // key puede ser 'sillas', 'mesas', 'armarios'
    };
    transactions: Transaction[];
}
 
/**
 * Gestiona el stock, operaciones de bsqueda y modificacin, y las transacciones de una tienda de muebles.
 * Implementa el patrn Singleton para asegurar una nica instancia de la clase.
 */
export class Stock {
    /**
     * La base de datos para almacenar el estado de la tienda.
     */
    private db: Low<DbSchema>;
    /**
     * La nica instancia de la clase Stock.
     */
    private static instance: Stock;
 
 
    /**
     * Constructor privado para prevenir la creacin directa de instancias y facilitar el patrn Singleton.
     */
    private constructor(dbPath:string) {
        const adapter = new JSONFile<DbSchema>(dbPath);
    
        this.db = new Low(adapter);
        this.initializeDb();
    }
    
    /**
     * Obtiene la instancia nica de la clase Stock, crendola si no existe.
     * @returns La instancia nica de la clase Stock.
     */
    public static getInstance(dbPath:string = 'db2.json'): Stock {
        if (!Stock.instance) {
            Stock.instance = new Stock(dbPath);
        }
        return Stock.instance;
    }
 
    /**
     * Inicializa la base de datos con estructuras de datos por defecto si est vaca.
     */
    private async initializeDb() {
        await this.db.read();
        this.db.data ||= { sillas: [], mesas: [], armarios: [], proveedores: [], clientes: [], stock: {}, transactions: [] };
        await this.db.write();
    }
 
    /**
     * Verifica si un ID es nico entre todas las entidades de muebles.
     * @param id El ID a verificar.
     * @returns True si el ID es nico, false de lo contrario.
     */
    private async idEsUnico(id: number) {
        await this.db.read(); 
        return !(this.db.data?.armarios.some(mueble => mueble.id === id) || this.db.data?.sillas.some(mueble => mueble.id === id) || this.db.data?.mesas.some(mueble => mueble.id === id));
    }
 
    /**
     * Aade un armario a la base de datos y actualiza el stock.
     * @param mueble El armario a aadir.
     */
    public async addArmario(mueble: Armario) {
        this.db.data?.armarios.push(mueble);
        this.db.data!.stock.armarios[mueble.id] = 10;
        await this.db.write();
    }
    
    /**
     * Aade una silla a la base de datos y actualiza el stock.
     * @param mueble La silla a aadir.
     */
    public async addSilla(mueble: Silla) {
        this.db.data?.sillas.push(mueble);
        this.db.data!.stock.sillas[mueble.id] = 10;
        await this.db.write();
    }
    
    /**
     * Aade una mesa a la base de datos y actualiza el stock.
     * @param mueble La mesa a aadir.
     */
    public async addMesa(mueble: Mesa) {
        this.db.data?.mesas.push(mueble);
        this.db.data!.stock.sillas[mueble.id] = 10;
        await this.db.write();
    }
 
    /**
     * Elimina una silla de la base de datos y actualiza el stock.
     * @param id El ID de la silla a eliminar.
     */
    public async deleteSilla(id: number) {
        if (this.db.data) {
            this.db.data.sillas = this.db.data.sillas.filter(mueble => mueble.id !== id);
            delete this.db.data!.stock.sillas[id];
            await this.db.write();
        }
    }
 
    /**
     * Elimina una mesa de la base de datos y actualiza el stock.
     * @param id El ID de la mesa a eliminar.
     */
    public async deleteMesa(id: number) {
        if (this.db.data) {
            this.db.data.mesas = this.db.data.mesas.filter(mueble => mueble.id !== id);
            delete this.db.data!.stock.mesas[id];
            await this.db.write();
        }
    }

     * Elimina un armario de la base de datos y actualiza el stock.
     * @param id El ID del armario a eliminar.
     */
    public async deleteArmario(id: number) {
        if (this.db.data) {
            this.db.data.armarios = this.db.data.armarios.filter(mueble => mueble.id !== id);
            delete this.db.data!.stock.armarios[id];
            await this.db.write();
        }
    }

     * Modifica una mesa existente reemplazndola con los datos de una nueva mesa.
     * Primero elimina la mesa existente con el ID proporcionado, luego aade la nueva mesa.
     * @param mueble La nueva mesa con la que se actualizar el registro.
     * @param id El ID de la mesa a modificar.
     */
    public async modifyMesa(mueble: Mesa, id:number) {
        this.deleteMesa(id);
        this.addMesa(mueble);
    }

    public async modifyArmario(mueble: Armario, id:number) {

    /**
     * Filtra y ordena una lista de muebles segn los criterios especificados por el usuario.
     * Se aplica un filtro basado en un criterio y un valor clave ingresados por el usuario. La comparacin es insensible a maysculas y minsculas.
     * Posteriormente, se ordenan los muebles filtrados segn el criterio de ordenacin y la direccin especificados (ascendente o descendente).
     * Este mtodo es utilizado por el mtodo buscarMueble() para obtener los muebles filtrados y ordenados.
     * @param tipo Tipo de mueble a filtrar ('sillas', 'mesas', 'armarios').
     * @param criterio Criterio por el cual se filtrar (por ejemplo, 'nombre' o 'material').
     * @param clave Valor clave para el filtro.
     * @param ordenacion Criterio de ordenacin ('precio' o 'descripcion').
     * @param asc Direccin de la ordenacin ('ascendente' o 'descendente').
     * @returns Una promesa que resuelve a una lista de muebles filtrados y ordenados.
     */
    private async filtrarMuebles(tipo: 'sillas' | 'mesas' | 'armarios', criterio: string, clave: string, ordenacion: string, asc: string): Promise<Mueble[]> {
        await this.db.read();
        const muebles = this.db.data?.[tipo] ?? [];
        // Convertir clave a minsculas para hacer la comparacin insensible a maysculas y minsculas
        const claveMinusc = clave.toLowerCase();
        // Asegurarse de que el criterio es una clave vlida para el tipo de mueble
        const mueblesFiltrados = muebles.filter(mueble => {
            // Utilizar asercin de tipo para acceder a la propiedad del objeto  (quebadero de cabeza esto)
            const valor = mueble[criterio as keyof typeof mueble];
            if (typeof valor === 'string') {
            // Convertir el valor a minsculas antes de la comparacin
            return valor.toLowerCase().includes(claveMinusc);
            }
            return false; // Si la propiedad no es una cadena, no incluir el mueble
        });

        // Ordenar segn el criterio de ordenacin
        if (ordenacion === 'precio') {
            mueblesFiltrados.sort((a, b) => asc === 'ascendente' ? a.precio - b.precio : b.precio - a.precio);
        } else { 
            mueblesFiltrados.sort((a, b) => asc === 'ascendente' ? a.descripcion.localeCompare(b.descripcion) : b.descripcion.localeCompare(a.descripcion));
        }

        return mueblesFiltrados;
    }

     * @returns Un objeto Date que representa la fecha proporcionada.
     */
    private convertirAFecha(fechaStr: string): Date {
        const [dia, mes, ao] = fechaStr.split('/').map(part => parseInt(part));

    private fechaEnRango(fecha: string, fechaInicio: Date, fechaFin: Date): boolean {
